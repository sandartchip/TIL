			(SELECT LV
			FROM(
				SELECT LV, NVL(PPLN_WGTNO, 0)
				FROM(
					SELECT ANLY_SRVR_USE_CNTNT, SUM(PPLN_WGTNO) PPLN_WGTNO
					FROM T_PROJECT A, T_PIPELINE B
					<![CDATA[
					WHERE A.ANLY_PPLNID = B.ANLY_PPLNID AND A.ANLY_JOB_STTS_CD IN ('0001', '0002', '0003')
						AND A.DEL_YN = 'N'
					]]>
					GROUP BY ANLY_SRVR_USE_CNTNT
				)A, (SELECT 'SER' || LEVEL LV FROM DUAL CONNECT BY LEVEL <![CDATA[<=]]> 2)B
				WHERE A.ANLY_SRVR_USE_CNTNT(+) = B.LV
				ORDER BY NVL(PPLN_WGTNO, 0)
			)WHERE ROWNUM = 1), 'SER1')
		FROM DUAL
	</select>
